def isPrime(n):
    count=0
    if(n==1):
        return "notprime"
    for i in range(2,n):
        if(n%i==0):
            count+=1
    if(count==0):
        return "prime"
    else:
        return "not prime"
def isTwin(n):
    for i in range(0,n):
        if((isPrime(n)=="prime" ) & (isPrime(n+2)=="prime" ) & (isPrime(n+1)=="not prime")):
            return True
        else:
            return False

import math
def isPerfect(n):
    c=1
    i=2
    sq=int(math.sqrt(n))+1
    for i in range(2,sq):
        if(n%i==0):
            c+=i
            c+=n//i
        i+=1       
    if(c==n):
        return "perfect number"
    else:
        return "not perfect number"

def lcm(m,n):
    if(m%n==0):
        return m
    if(n%m==0):
        return n
    c=max(m,n)
    while(1):
        if(c%m==0 and c%n==0):
            return c
        c+=1
